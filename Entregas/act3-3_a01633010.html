Escribe la notación BNF y EBNF para la gramática necesaria para definir modulos y funciones en Elixir.

BNF


<module> ::= <module-macro> <module-name> <do-operator> <EOL> <functions> <EOL> <end-operator> <EOL>

<functions> ::= <function> | <function> <EOL> <functions>

<function> ::= <function-macro> <function-name> <parameters> <do-operator> <expressions> <end-operator>

<expressions> ::= <expression> | <expression> <EOL> <expressions>

<expression> ::= <EOL>

<function-macro> ::= <public-function-macro> | <private-function-macro>

<parameters> ::= <opening-parenthesis> <parameter> <closing-parenthesis> | <opening-parenthesis> <closing-parenthesis>

<parameter> ::= | <parameter> <separator> <parameter>

<module-macro> ::= “defmodule”

<public-function-macro> ::= “def”

<private-function-macro> ::= “defp”

<opening-parenthesis> ::= “(“

<closing-parenthesis> ::= “)”

<separator> ::= “,”

<do-operator> ::= “do”

<end-operator> ::= “end”

<EOL> ::= “\n”

EBNF

<module> ::= “defmodule” <EOL> <module-name> <EOL> “do” { <function> + } “end” <EOL>;

<function> ::= (“def” | “defp”) <function-name> “(“ { <parameter> * } ? “)” “do” { <expression> +} <EOL>;

<parameter> ::= <id> [ “,” ];

<expression> ::= ? Elixir-specified expression rule ?;

<function-name> ::= ? Elixir-specified naming rule ?;

<module-name> ::= ? Elixir-specified naming rule ?;

<id> ::= ? Elixir-specified id name rule ?;

<EOL> ? System-specific end-of-line character ?;
